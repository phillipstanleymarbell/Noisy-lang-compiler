274 /*
275 * For a dimensional matrix with n parameters (n columns) and rank r,
276 * (i.e., equivalently, with r pivot columns ==> r linearly-independent
277 * columns) there are choose(n, r) ways in which we can rearrange the
278 * columns of the dimensional matrix to yield a unique Pi group (see
279 * Harald Hanche-Olsen, 2004 and E. Buckingham, 1914).
280 *
281 * For each of these orderings of the r pivot columns, we generate
282 * all the possible n-bit words which have r '1's and we place the
283 * r pivot column indices at those positions.
284 *
285 * If we also wanted to account not just for placement of the pivot
286 * columns, but also for their permutations relative to each other,
287 * we could use Heap's algorithm (B. R. Heap, "Permutations by
288 * interchanges". The Computer Journal. 6 (3): 2934) to find all the
289 * r! permutations of the r pivot column indices.
290 *
291 * Limitations:
292 * (1) We assume inputMatrix has less than 64 columns as we use a
293 * uint64_t for the pivot position bitmap.
294 */

615 canonicallyReorderedLabels[countKernel] = (char **)calloc(dimensionalMatrixColumnCount, sizeof(char *));
616 for (int countRow = 0; countRow < dimensionalMatrixColumnCount; countRow++)
617 {
618 /*
619 * The labels are not lexicographically reordered yet.
620 * They are only reordered to match the permutation for
621 * each kernel.
622 */
623 canonicallyReorderedLabels[countKernel][countRow] =
624 (char *)calloc(strlen(dimensionalMatrixColumnLabels[permutedIndexArrayPointer[countKernel * dimensionalMatrixColumnCount + countRow]]), sizeof(char));
625 canonicallyReorderedLabels[countKernel][countRow] =
626 dimensionalMatrixColumnLabels[permutedIndexArrayPointer[countKernel * dimensionalMatrixColumnCount + countRow]];
627 }
628 }
629 /*
630 * Now sort the labels in lexicographic order (dictionary order)
631 */
632 int temp;
633 for (int countKernel = 0; countKernel < *numberOfUniqueKernels; countKernel++)
634 {
635 tmpPosition[countKernel] = (int *)calloc(dimensionalMatrixColumnCount, sizeof(int));
636 for (int i = 0; i < dimensionalMatrixColumnCount; i++)
637 {
638 /*
639 * Reset tmpPosition so that it could be used for the re-ordering
640 * of invariant->nullSpace.
641 */
642 tmpPosition[countKernel][i] = i;
643 }
644 for (int i = 0; i < (dimensionalMatrixColumnCount -1); i++)
645 {
646 for (int j = i + 1; j < dimensionalMatrixColumnCount; j++)
647 {
648 if (strcmp(canonicallyReorderedLabels[countKernel][i], canonicallyReorderedLabels[countKernel][j]) > 0)
649 {
650 /*
651 * Update the temporary position pointer and in the mean time
652 * sort the label for future debug use. The labels for all kernels
653 * should be lexicographically ordered after this step, so they
654 * should all be the same.
655 */
656 temp = tmpPosition[countKernel][i];
657 tmpPosition[countKernel][i] = tmpPosition[countKernel][j];
658 tmpPosition[countKernel][j] = temp;
659
660 char * tempString;
661 /*
662 * NOTE: We are passing the string pointer, not copying the string.
663 */
664 tempString = canonicallyReorderedLabels[countKernel][i];
665 canonicallyReorderedLabels[countKernel][i] = canonicallyReorderedLabels[countKernel][j];
666 canonicallyReorderedLabels[countKernel][j] = tempString;
667
668 int tempIndex;
669 tempIndex = permutedIndexArrayPointer[countKernel * dimensionalMatrixColumnCount + i];
670 permutedIndexArrayPointer[countKernel * dimensionalMatrixColumnCount + i]
671 = permutedIndexArrayPointer[countKernel * dimensionalMatrixColumnCount + j];
672 permutedIndexArrayPointer[countKernel * dimensionalMatrixColumnCount + j] = tempIndex;
673 }
674 }
675 }
676 }
677 /*
678 * The following codes could be updated to exploit the ease
679 * of implementation using Eigen libraries.
680 */
681 double *** reorderedNullSpace = (double ***)calloc(*numberOfUniqueKernels, sizeof(double **));
682 for (int countKernel = 0; countKernel < *numberOfUniqueKernels; countKernel++)
683 {
684 reorderedNullSpace[countKernel] = (double **)calloc(kernelColumnCount, sizeof(double *));
685 for (int countColumn = 0; countColumn < kernelColumnCount; countColumn++)
686 {
687 reorderedNullSpace[countKernel][countColumn] = (double *)calloc(dimensionalMatrixColumnCount, sizeof(double));
688 for (int countRow = 0; countRow < dimensionalMatrixColumnCount; countRow++)
689 {
690 /*
691 * NOTE: invariant->nullspace is stored as [kernel][row][column], whereas
692 * reorderedNullSpace and all that follows are [kernel][column][row]
693 */
694 reorderedNullSpace[countKernel][countColumn][countRow] = nullSpace[countKernel][tmpPosition[countKernel][countRow]][countColumn];
695 }
696 }
697 }
698
699 /*
700 * Check for zero value to decide the smallest lexicographical
701 * component to be used as the base for each specific column.
702 */
703 bool firstNonZeroFound = false;
704 double ** factor = (double **)calloc(*numberOfUniqueKernels,sizeof(double *));
705 for (int countKernel = 0; countKernel < *numberOfUniqueKernels; countKernel++)
706 {
707 for (int countColumn = 0; countColumn < kernelColumnCount; countColumn++)
708 {
709 factor[countKernel] = (double *)calloc(kernelColumnCount,sizeof(double));
710 for (int countRow = 0; countRow < dimensionalMatrixColumnCount; countRow++)
711 {
712 if (reorderedNullSpace[countKernel][countColumn][countRow] != 0 && !firstNonZeroFound)
713 {
714 factor[countKernel][countColumn] = 1 / reorderedNullSpace[countKernel][countColumn][countRow];
715 reorderedNullSpace[countKernel][countColumn][countRow] = 1;
716 firstNonZeroFound = true;
717 continue;
718 }
719 if (firstNonZeroFound)
720 {
721 /*
722 * This makes sure that each element within a column is
723 * refactorized using the factor derived from the first
724 * none zero row element.
725 */
726 reorderedNullSpace[countKernel][countColumn][countRow] =
727 reorderedNullSpace[countKernel][countColumn][countRow] * factor[countKernel][countColumn];
728 }
729 }
730 firstNonZeroFound = false;
731 }
732 }
733
734 *canonicalLabels = canonicallyReorderedLabels;
735
736 free(factor);
737 free(tmpPosition);
738
739 return reorderedNullSpace;
740 }

742 double ***
743 newtonEigenLibraryInterfaceSortedCanonicallyReorderedPiGroups(double *** nullSpaceRowReordered,
744 char *** canonicallyReorderedLabels,
745 int kernelColumnCount, int dimensionalMatrixColumnCount,
746 int * numberOfUniqueKernels)
747 {
748 /*
749 * We create the matrices in Eigen by mapping the elements
750 * from nullSpaceRowReordered.
751 */
752 ColMajorOrderMatrixXd *eigenInterfaceReorderKernels = new ColMajorOrderMatrixXd[*numberOfUniqueKernels];
753
754 for (int countKernel = 0; countKernel < *numberOfUniqueKernels; countKernel++)
755 {
756 ColMajorOrderMatrixXd temp(dimensionalMatrixColumnCount, kernelColumnCount);
757 for (int countColumn = 0; countColumn < kernelColumnCount; countColumn++)
758 {
759 for (int countRow = 0; countRow < dimensionalMatrixColumnCount; countRow++)
760 {
761 temp(countRow, countColumn) = nullSpaceRowReordered[countKernel][countColumn][countRow];
762 }
763 }
764
765 eigenInterfaceReorderKernels[countKernel] = temp;
766 }
767
768 /*
769 * Compare all the columns and sort the columns (representing pi's)
770 * in the following lexicographic order:
771 * Read out pi_i, form the nonzero labels as a combined string
772 * and then compare the string lexicographically.
773 */
774
775 char *** string = (char ***)calloc(*numberOfUniqueKernels, sizeof (char **));
776 int * whichRow = (int *)calloc(dimensionalMatrixColumnCount, sizeof(int));
777 int stringLength;
778
779 int nonZeroRowCount = 0;
780
781 for (int countKernel = 0; countKernel < *numberOfUniqueKernels; countKernel++)
782 {
783 string[countKernel] = (char **)calloc(kernelColumnCount, sizeof (char *));
784 for (int countColumn = 0; countColumn < kernelColumnCount; countColumn++)
785 {
786 stringLength = 0;
787 nonZeroRowCount = 0;
788 for (int countRow = 0; countRow < dimensionalMatrixColumnCount; countRow++)
789 {
790 if (eigenInterfaceReorderKernels[countKernel](countRow, countColumn) != 0)
791 {
792 /*
793 * Assign the row index and the length of the combined string.
794 */
795 whichRow[nonZeroRowCount++] = countRow;
796 stringLength += (strlen(canonicallyReorderedLabels[countKernel][countRow]) + 1);
797 }
798 }
799
800 if (stringLength != 0)
801 {
802 string[countKernel][countColumn] = (char *)calloc(stringLength, sizeof(char));
803 /*
804 * Copy the first non-zero parameter label to 'string'.
805 */
806 strcpy(string[countKernel][countColumn], canonicallyReorderedLabels[countKernel][whichRow[0]]);
807 for (int i = 1; i < nonZeroRowCount; i++)
808 {
809 /*
810 * Attach (concatenate) the following labels.
811 */
812 strcat(string[countKernel][countColumn], canonicallyReorderedLabels[countKernel][whichRow[i]]);
813 }
814 }
815 }
816
817 /*
818 * Now we loop through all the columns and compare the strings
819 * then reorder them in lexicographic order.
820 */
821 for (int i = 0; i < (kernelColumnCount - 1); i++)
822 {
823 for (int j = i + 1; j < kernelColumnCount; j++)
824 {
825 if (string[countKernel][i] == NULL__null || string[countKernel][j] == NULL__null)
826 {
827 continue;
828 }
829 if (strcmp(string[countKernel][i],string[countKernel][j]) > 0)
830 {
831 eigenInterfaceReorderKernels[countKernel].col(i).swap(eigenInterfaceReorderKernels[countKernel].col(j));
832
833 char * tempString;
834 tempString = string[countKernel][i];
835 string[countKernel][i] = string[countKernel][j];
836 string[countKernel][j] = tempString;
837 }
838 }
839 }
840 }
841
842 /*
843 * Reform the canonically ordered nullspace
844 */
845 double *** reorderedNullSpace = (double ***)calloc(*numberOfUniqueKernels, sizeof(double **));
846 for (int countKernel = 0; countKernel < *numberOfUniqueKernels; countKernel++)
847 {
848 reorderedNullSpace[countKernel] = (double **)calloc(kernelColumnCount, sizeof(double *));
849 for (int countColumn = 0; countColumn < kernelColumnCount; countColumn++)
850 {
851 reorderedNullSpace[countKernel][countColumn] = (double *)calloc(dimensionalMatrixColumnCount, sizeof(double));
852 for (int countRow = 0; countRow < dimensionalMatrixColumnCount; countRow++)
853 {
854 reorderedNullSpace[countKernel][countColumn][countRow] = eigenInterfaceReorderKernels[countKernel](countRow, countColumn);
855 }
856 }
857 }
858
859 free(string);
860 free(whichRow);
861 return reorderedNullSpace;
862 }

864 double ***
865 newtonEigenLibraryInterfaceWeedOutDuplicatePiGroups(double *** nullSpace,
866 double *** nullSpaceCanonicallyReordered,
867 int kernelColumnCount,
868 int dimensionalMatrixColumnCount,
869 int * numberOfUniqueKernels,
870 int * numberOfTotalKernels)
871 {
872 /*
873 * We create the matrices in Eigen by mapping the elements
874 * from nullSpaceCanonicallyReordered.
875 */
876 ColMajorOrderMatrixXd *eigenInterfaceReorderKernels = new ColMajorOrderMatrixXd[*numberOfUniqueKernels];
877
878 for (int countKernel = 0; countKernel < *numberOfUniqueKernels; countKernel++)
879 {
880 ColMajorOrderMatrixXd temp(dimensionalMatrixColumnCount, kernelColumnCount);
881 for (int countColumn = 0; countColumn < kernelColumnCount; countColumn++)
882 {
883 for (int countRow = 0; countRow < dimensionalMatrixColumnCount; countRow++)
884 {
885 temp(countRow, countColumn) = nullSpaceCanonicallyReordered[countKernel][countColumn][countRow];
886 }
887 }
888 eigenInterfaceReorderKernels[countKernel] = temp;
889 }
890
891 *numberOfTotalKernels = *numberOfUniqueKernels;
892
893 int kernelToBeDeprecatedCount = 0;
894 int * kernelToBeDeprecated = (int *)calloc(*numberOfTotalKernels, sizeof(int));
895 bool isDeprecated = false;
896
897 for (int k = 0; k < *numberOfTotalKernels; k++)
898 {
899 /*
900 * Initiate the kernelToBeDeprecated array
901 */
902 kernelToBeDeprecated[k] = -1;
903 }
904
905 for (int i = 0; i < (*numberOfTotalKernels - 1); i++)
906 {
907 for (int k = 0; k < kernelToBeDeprecatedCount; k++)
908 {
909 if (i == kernelToBeDeprecated[k])
910 {
911 isDeprecated = true;
912 break;
913 }
914 }
915 if (isDeprecated)
916 {
917 isDeprecated = false;
918 continue;
919 }
920 else
921 {
922 for (int j = i + 1; j < *numberOfTotalKernels; j++)
923 {
924 /*
925 * We subtract the two kernels, element-wise, to see
926 * if the result is a zero matrix.
927 */
928
929 for (int k = 0; k < kernelToBeDeprecatedCount; k++)
930 {
931 if (j == kernelToBeDeprecated[k])
932 {
933 isDeprecated = true;
934 }
935 }
936 if (isDeprecated)
937 {
938 /*
939 * There is no need to check the kernel that has already
940 * been deprecated. We skip this j-th kernel and continue
941 * to the next.
942 */
943 isDeprecated = false;
944 continue;
945 }
946 else
947 {
948 if ( (eigenInterfaceReorderKernels[i] - eigenInterfaceReorderKernels[j]).isZero())
949 {
950 *numberOfUniqueKernels -= 1;
951 kernelToBeDeprecated[kernelToBeDeprecatedCount++] = j;
952 }
953 }
954
955 }
956 }
957 }
958
959 /*
960 * Now arrange the kernelToBeDeprecated array in ascending order.
961 */
962 int tempKernelIndex = -1;
963 for (int i = 0; i < kernelToBeDeprecatedCount - 1; i++)
964 {
965 for (int j = i + 1; j < kernelToBeDeprecatedCount; j ++)
966 {
967 if (kernelToBeDeprecated[i] > kernelToBeDeprecated[j])
968 {
969 tempKernelIndex = kernelToBeDeprecated[i];
970 kernelToBeDeprecated[i] = kernelToBeDeprecated[j];
971 kernelToBeDeprecated[j] = tempKernelIndex;
972 }
973 }
974 }
975
976 /*
977 * Reform the nullspace
978 */
979 double *** reorderedNullSpace = (double ***)calloc(*numberOfUniqueKernels, sizeof(double **));
980 int reorderedKernelCount = 0;
981 kernelToBeDeprecatedCount = 0;
982
983 for (int countKernel = 0; countKernel < *numberOfTotalKernels; countKernel++)
984 {
985 if(countKernel == kernelToBeDeprecated[kernelToBeDeprecatedCount])
986 {
987 kernelToBeDeprecatedCount++;
988 }
989 else
990 {
991 /*
992 * Increase this count only when there is no duplicate
993 */
994 reorderedNullSpace[reorderedKernelCount] = (double **)calloc(kernelColumnCount, sizeof(double *));
995 for (int countColumn = 0; countColumn < kernelColumnCount; countColumn++)
996 {
997 reorderedNullSpace[reorderedKernelCount][countColumn] = (double *)calloc(dimensionalMatrixColumnCount, sizeof(double));
998 for (int countRow = 0; countRow < dimensionalMatrixColumnCount; countRow++)
999 {
1000 reorderedNullSpace[reorderedKernelCount][countColumn][countRow] = eigenInterfaceReorderKernels[countKernel](countRow,countColumn);
1001 }
1002 }
1003 reorderedKernelCount++;
1004 }
1005 }
1006
1007 /*
1008 * In order to reuse irPass-dimensionalMatrixKernelPrinter
1009 * the following changes need to be made accordingly:
1010 * Update invariant->nullSpace using nullSpaceWithoutDuplicates.
1011 * Free the rest of the nullSpace.
1012 */
1013
1014 for (int countKernel = 0; countKernel < *numberOfUniqueKernels; countKernel++)
1015 {
1016 for (int countRow = 0; countRow < dimensionalMatrixColumnCount; countRow++)
1017 {
1018 for (int countColumn = 0; countColumn < kernelColumnCount; countColumn++)
1019 {
1020 /*
1021 * NOTE: invariant->nullspace is stored as [kernel][row][column], whereas
1022 * reorderedNullSpace and all that follows are [kernel][column][row]
1023 */
1024 nullSpace[countKernel][countRow][countColumn] = reorderedNullSpace[countKernel][countColumn][countRow];
1025 }
1026 }
1027 }
1028
1029 /*
1030 * Free up the memory which will no longer be used
1031 * Since the duplicates have been weeded out.
1032 */
1033 for (int countKernel = *numberOfUniqueKernels; countKernel < *numberOfTotalKernels; countKernel++)
1034 {
1035 free(nullSpace[countKernel]);
1036 }
1037
1038 free(kernelToBeDeprecated);
1039
1040 return reorderedNullSpace;
1041 }
